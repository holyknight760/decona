#알파 diversity에서 그룹간의 다른지를 확인하고 싶은 경우
#https://docs.qiime2.org/2022.2/tutorials/pd-mice/
#Faith's Phylogenetic Diversity
$ qiime diversity alpha-group-significance \
  --i-alpha-diversity ./diversity/faith_pd_vector.qza \
  --m-metadata-file metadata_16s.tsv \
  --o-visualization ./diversity/faiths_pd_statistics.qzv \
  --verbose
#Saved Visualization to: ./diversity/faiths_pd_statistics.qzv

#베타 diversity 하기(PERMANOVA test)
#주의 meta data에서 특정 column을 정해줘야함
$ time qiime diversity beta-group-significance \
--i-distance-matrix ./diversity/bray_curtis_distance_matrix.qza \
--m-metadata-file metadata_16s.tsv \
--m-metadata-column species \
--o-visualization ./diversity/bray_curtis_distance-depth-significance.qzv \
--p-pairwise \
--verbose
